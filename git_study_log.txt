## git添加文件
git add .

## git 删除工作区的文件
git rm --cached git_study_log.txt

## git 提交文件
git commit -m "初始化文件"

## 查看git 的提交记录
git reflog

## git log的详细日志
git log

## git 版本回滚
git reset --hard 232232

## git 创建分支
git branch 分支名称

## git 切换分支
git checkout 分支名称

## git 删除分支
git branch -d 分支名称

## git 强制删除分支
git branch -D 分支名称

## git 合并分支
git merge 源分支

## 拉取远程的代码
git pull
